

1 - join
2 - nick
3 - pass
4 - user
5 - kick
6 - invite
7 - mode
8 - topic

1
{
	// multiple channels and also multiple keys

	// 1.1
	join #channel1
	// 1.2
	join #channel1,#channel2
	// 1.3
	join #channel1,#channel2 key1,key2
	// 1.4
	join #channel1,#channel2 key1,key2
}

2
{
	// multiple nicks
	// 2.1
	nick nick1
	// 2.2
	nick nick1,nick2
}

3
{
	// 3.1
	pass password1
}

4
{
	user user1 user2 user3 user4
}

5
{
	// 5.1
	kick #channel1 nick1
	// 5.2
	kick #channel1 nick1 :reason
}

6
{
	// 6.1
	invite nick1 #channel1
}

7
{
	// mode i o t k l
	// 7.1
	mode #channel1 +i
	// 7.2
	mode #channel1 +o nick1
	// 7.3
	mode #channel1 +t
	// 7.4
	mode #channel1 +k key1
	// 7.5
	mode #channel1 +l limit
}

8
{
	// 8.1
	topic #channel1 :topic
	// 8.2
	topic #channel1
}
